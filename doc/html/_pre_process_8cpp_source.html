00004 void Tokenize(const string& str, vector<string>& tokens,
00005                       const string& delimiters){
00006     // Skip delimiters at beginning.
00007     string::size_type lastPos = str.find_first_not_of(delimiters, 0);
00008     // Find first "non-delimiter".
00009     string::size_type pos     = str.find_first_of(delimiters, lastPos);
00010     while (string::npos != pos || string::npos != lastPos)  {
00011         // Found a token, add it to the vector.
00012         tokens.push_back(str.substr(lastPos, pos - lastPos));
00013         // Skip delimiters.  Note the "not_of"
00014         lastPos = str.find_first_not_of(delimiters, pos);
00015         // Find next "non-delimiter"
00016         pos = str.find_first_of(delimiters, lastPos);
00017     }
