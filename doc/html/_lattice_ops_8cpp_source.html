00049 IOSet *Prime(NCluster *a, RelationGraph *g, int s, int t,int min){
00050     assert(g->IsEdge(s,t));    
00051     IOSet *sSet = a->GetSetById(s);
00052     if(min < 0) min=0;
00053     Context *ctx = g->GetContext(s,t);
00054     if(a->GetSetById(s)->Size() == 0) return NULL; //since 0 return as is
00055     else{                            //compute the prime as set intersections (this is a fairly naive method)
00056     
00057         IOSet *rslt = new IOSet(ctx->GetSet(s,sSet->At(0)));
00058         if(rslt->Size() < min){  //if sSet only has a single object then perform min check here
00059             delete rslt;
00060             return NULL;
00061         }
00062         for(int i=1; i < sSet->Size(); i++){
00063             if(rslt->Size() < min ||ctx->GetSet(s,sSet->At(i))->Size() < min ){ // no point in performing intersection if size < min
00064                 delete rslt;
00065                 return NULL;
00066             }
00067             IOSet *tmp = rslt;
00068             rslt = Intersect(rslt,ctx->GetSet(s,sSet->At(i)));
00069             delete tmp;
00070             tmp = NULL;
00071         }
00072         if( rslt->Size() >= min) {//final check for satasfaction
00073             rslt->SetId(t);
00074             return rslt;
00075         }else{
00076             delete rslt;
00077             return NULL;
00078         }
00079     }
